// DataviewJS

const root = "tomosidian/01_日記"; // 日記ルートフォルダ
const today = window.moment().format("YYYY-MM-DD");
const monthFolder = window.moment().format("YYYY年MM月");
const dailyFileName = `${today}.md`;
const dailyFilePath = `${root}/${monthFolder}/${dailyFileName}`;
const vault = app.vault;

// ==== ▼ ページを開いた時に自動でファイル作成（存在しなければ） ====
(async () => {
  const folderPath = `${root}/${monthFolder}`;

  // 月フォルダが存在しなければ作成
  if (!await vault.adapter.exists(folderPath)) {
    await vault.createFolder(folderPath);
    new Notice(`フォルダを作成しました: ${folderPath}`);
  }

  // 「2025-10-30～.md」で始まるファイルが存在するか確認
  const existingFiles = vault.getFiles()
    .filter(f => f.path.startsWith(`${folderPath}/${today}`) && f.path.endsWith(".md"));

  // 同日ファイルが一つも無い場合のみ自動作成
  if (existingFiles.length === 0) {
    await vault.create(dailyFilePath, "");
    new Notice(`今日の日記を自動作成しました: ${dailyFileName}`);
  
// ▼▼ ここを追加：新規ファイルを新しいタブで開く ▼▼
	const file = vault.getAbstractFileByPath(dailyFilePath);
    if (file) {
      const leaf = app.workspace.getLeaf("tab"); // true で新しいタブを作る
      await leaf.openFile(file);
    }
    // ▲▲ 追加ここまで ▲▲
    }
})();

// ==== ▼ ボタンを最上部に表示 ====
const btnContainer = this.container.createEl("div", { cls: "create-today-container" });
const btn = btnContainer.createEl("button", { text: `今日の日記を新規作成 (${today})` });
btn.style.margin = "0.5em 0 1em 0";
btn.style.padding = "6px 12px";
btn.style.borderRadius = "6px";
btn.style.cursor = "pointer";
btn.style.backgroundColor = "#556b8e";
btn.style.color = "#f5f5f5";
btn.style.fontWeight = "300";
btn.style.letterSpacing = "1px";
btn.style.border = "none";

// ==== ▼ ボタンクリック時の処理（常に新しいファイルを作る） ====
btn.onclick = async () => {
  const folderPath = `${root}/${monthFolder}`;

  // 月フォルダが存在しなければ作成
  if (!await vault.adapter.exists(folderPath)) {
    await vault.createFolder(folderPath);
    new Notice(`フォルダを作成しました: ${folderPath}`);
  }

  // 既存の同日ファイル数を確認して連番をつける
  const existingFiles = vault.getFiles()
    .filter(f => f.path.startsWith(`${folderPath}/${today}`) && f.path.endsWith(".md"));

  let newFileName;
  if (existingFiles.length === 0) {
    newFileName = `${today}.md`;
  } else {
    newFileName = `${today} (${existingFiles.length + 1}).md`;
  }

  const newFilePath = `${folderPath}/${newFileName}`;
  await vault.create(newFilePath, "");
  new Notice(`新しい日記を作成しました: ${newFileName}`);

  // 新規ファイルを開く
  const file = vault.getAbstractFileByPath(newFilePath);
  if (file) await app.workspace.getLeaf("tab").openFile(file);
};

// ==== ▼ 日記一覧を表示 ====
const pages = dv.pages(`"${root}"`)
  .where(p => p.file.name.match(/^\d{4}-\d{2}-\d{2}/))
  .sort(p => p.file.name, 'desc');

dv.header(2, "📅 日記一覧");
for (let page of pages) {
  const link = dv.fileLink(page.file.path, false, page.file.name);
  dv.paragraph(`- ${link}`);
}
