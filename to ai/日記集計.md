```dataviewjs
const root = "tomosidian/01_日記";
const container = dv.container;

// ===== ▼ 即時表示（最重要） =====
container.innerHTML = `
<div id="status" style="opacity:0.8; margin-bottom:0.5em;">
  📂 読み込み準備中…
</div>
<div id="table-area"></div>
`;

// UIを先に描画させる
await new Promise(r => setTimeout(r, 50));

const statusEl = container.querySelector("#status");
const tableArea = container.querySelector("#table-area");

// 対象ページ取得（件数把握）
const allPages = dv.pages()
    .where(p => p.file.path.startsWith(root))
    .array();

const total = allPages.length;
let processed = 0;

let monthly = {};

// ===== ▼ スタイル（このコード内で完結） =====
const style = document.createElement("style");
style.textContent = `
.month-table {
    width: 100%;
    border-collapse: collapse;
    margin-top: 1em;
}
.month-table th, .month-table td {
    border-bottom: 1px solid rgba(255,255,255,0.1);
    padding: 8px 10px;
}
.month-table th {
    text-align: left;
    background: rgba(255,255,255,0.08);
    font-weight: 600;
}
.month-table td {
    color: #e5e9f0;
}
`;
document.head.appendChild(style);

// ===== ▼ テーブル枠だけ先に作る =====
dv.header(2, "📘 月ごとの日記統計（件数・合計文字数・平均文字数）");

const tableEl = dv.el("table", "", { cls: "month-table", parent: tableArea });
tableEl.innerHTML = `
<tr>
  <th>月</th>
  <th>件数</th>
  <th>合計文字数</th>
  <th>平均文字数</th>
</tr>
`;

// ===== ▼ メイン処理 =====
for (const p of allPages) {
    processed++;

    statusEl.textContent = `📖 読み込み中… ${processed} / ${total} 件`;

    const name = p.file.name;
    const datePart = name.substring(0, 10);
    if (!/^\d{4}-\d{2}-\d{2}$/.test(datePart)) continue;

    const year = datePart.substring(0, 4);
    const month = datePart.substring(5, 7);
    const ym = `${year}-${month}`;

    if (!monthly[ym]) {
        monthly[ym] = { count: 0, chars: 0 };
    }

    monthly[ym].count++;

    const content = await dv.io.load(p.file.path);
    monthly[ym].chars += content.length;

    // 定期的にUIへ制御を返す（フリーズ防止）
    if (processed % 10 === 0) {
        await new Promise(r => setTimeout(r, 0));
    }
}

// ===== ▼ 集計結果を描画 =====
const rows = Object.entries(monthly)
    .sort((a, b) => b[0].localeCompare(a[0]));

for (const [ym, data] of rows) {
    const avg = data.count === 0 ? 0 : Math.round(data.chars / data.count);
    const tr = document.createElement("tr");
    tr.innerHTML = `
      <td>${ym}</td>
      <td>${data.count}</td>
      <td>${data.chars.toLocaleString()}</td>
      <td>${avg.toLocaleString()}</td>
    `;
    tableEl.appendChild(tr);
}

// ===== ▼ 完了表示 =====
statusEl.textContent = `✅ 読み込み完了（${total} 件処理）`;

```
